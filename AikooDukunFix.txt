-- Drag (main & bubble)
local dragTarget = nil; local dragging = false; local dragStart, startPos
main.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragTarget = main; dragging = true; dragStart = input.Position; startPos = main.Position
        input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then dragging = false end end)
    end
end)
bubble.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragTarget = bubble; dragging = true; dragStart = input.Position; startPos = bubble.Position
        input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then dragging = false end end)
    end
end)
UIS.InputChanged:Connect(function(input)
    if dragging and dragTarget and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local delta = input.Position - dragStart
        dragTarget.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
        if dragTarget == main then shadow.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X -5, startPos.Y.Scale, startPos.Y.Offset + delta.Y -5) end
    end
end)

-- ==================== ENGINE ====================
local enemiesFolder = nil
spawn(function()
    while not enemiesFolder do
        local alives = workspace:FindFirstChild("Alives")
        if alives then enemiesFolder = alives:FindFirstChild("Enemies") end
        if not enemiesFolder then
            for _, v in ipairs(workspace:GetDescendants()) do
                if v:IsA("Folder") and v.Name == "Enemies" and v.Parent and v.Parent.Name == "Alives" then
                    enemiesFolder = v; break
                end
            end
        end
        task.wait(0.5)
    end
end)

local function getHumanoid(obj)
    local m = obj
    if obj:IsA("BasePart") then m = obj.Parent end
    if m and (m:IsA("Model") or m:IsA("Folder")) then return m:FindFirstChildWhichIsA("Humanoid") end
end

-- ESP otomatis
local hlts = {}; local playerHlts = {}
local function clearESP()
    for _, hl in pairs(hlts) do hl:Destroy() end; hlts = {}
    for _, hl in pairs(playerHlts) do hl:Destroy() end; playerHlts = {}
end

local function updateESP()
    for model, hl in pairs(hlts) do if not model or not model.Parent then hl:Destroy(); hlts[model] = nil end end
    for char, hl in pairs(playerHlts) do if not char or not char.Parent or not Players:GetPlayerFromCharacter(char) then hl:Destroy(); playerHlts[char] = nil end end
    if not enemiesFolder then return end
    for _, v in ipairs(enemiesFolder:GetChildren()) do
        local hh = getHumanoid(v)
        if hh and hh.Health > 0 then
            if not hlts[v] then
                local hl = Instance.new("Highlight", v)
                hl.FillColor = Color3.fromRGB(180,180,180); hl.OutlineColor = Color3.new(1,1,1); hl.FillTransparency = 0.5
                hlts[v] = hl
            end
        elseif hlts[v] then hlts[v]:Destroy(); hlts[v] = nil end
    end
    for _, other in ipairs(Players:GetPlayers()) do
        if other ~= Player and other.Character then
            local char = other.Character
            if not playerHlts[char] then
                local hl = Instance.new("Highlight", char)
                hl.FillColor = Color3.fromRGB(0,255,0); hl.OutlineColor = Color3.new(1,1,1); hl.FillTransparency = 0.5
                playerHlts[char] = hl
            end
        end
    end
end

-- Auto Kill
local autoOn = false
local function killEn(d)
    if not d.Humanoid or d.Humanoid.Health <= 0 then return end
    local hh = d.Humanoid
    for _, tag in ipairs({"creator","killTag","source"}) do
        local obj = hh:FindFirstChild(tag) or Instance.new("ObjectValue", hh); obj.Name = tag; pcall(function() obj.Value = Player end)
    end
    pcall(function() hh:TakeDamage(hh.MaxHealth + 1) end)
    task.wait(0.1)
    if hh.Health > 0 then pcall(function() hh.Health = 0 end) end
end
local function killAll()
    if not enemiesFolder then return end
    for _, v in ipairs(enemiesFolder:GetChildren()) do
        local hh = getHumanoid(v)
        if hh and hh.Health > 0 then killEn({Model = v, Humanoid = hh}); task.wait(0.05) end
    end
end

-- Infinite Jump (optimized)
local infJumpEnabled = false
local infJumpConn = nil

local function enableInfJump()
    if infJumpConn then return end
    infJumpConn = UIS.JumpRequest:Connect(function()
        if not infJumpEnabled then return end
        local char = Player.Character
        if char then
            local hh = char:FindFirstChildWhichIsA("Humanoid")
            if hh then hh:ChangeState(Enum.HumanoidStateType.Jumping) end
        end
    end)
end

local function disableInfJump()
    if infJumpConn then infJumpConn:Disconnect(); infJumpConn = nil end
end

-- Auto Proximity Prompt (fireproximityprompt)
local autoPromptConn = nil
autoPromptConn = RunService.Heartbeat:Connect(function()
    local char = Player.Character; if not char then return end
    local root = char:FindFirstChild("HumanoidRootPart"); if not root then return end
    local bestDist, bestPrompt = 100, nil
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("ProximityPrompt") and obj.Parent then
            local parent = obj.Parent
            local pos = parent:IsA("BasePart") and parent.Position or (parent:IsA("Model") and parent.PrimaryPart and parent.PrimaryPart.Position)
            if pos then
                local dist = (root.Position - pos).Magnitude
                if dist <= 100 and dist < bestDist then bestDist = dist; bestPrompt = obj end
            end
        end
    end
    if bestPrompt then pcall(function() fireproximityprompt(bestPrompt) end) end
end)

-- Instant Proximity Prompt (PromptButtonHoldBegan)
local instantPromptConn = ProximityPromptService.PromptButtonHoldBegan:Connect(function(prompt, plr)
    if plr == Player then fireproximityprompt(prompt) end
end)

-- Matikan fitur prompt saat GUI dihancurkan
local function disableAllPrompts()
    if autoPromptConn then autoPromptConn:Disconnect(); autoPromptConn = nil end
    if instantPromptConn then instantPromptConn:Disconnect(); instantPromptConn = nil end
end

-- Loop utama (update ESP, count, auto kill) – dioptimalkan dengan jeda lebih besar
local loopActive = true
spawn(function()
    while loopActive do
        updateESP()
        if enemiesFolder then
            local count = 0
            for _, v in ipairs(enemiesFolder:GetChildren()) do
                local hh = getHumanoid(v)
                if hh and hh.Health > 0 then count = count + 1 end
            end
            countLabel.Text = "Enemies: " .. count
            if autoOn and count > 0 then killAll() end
        else
            countLabel.Text = "Enemies: ?"
        end
        task.wait(0.5) -- dari 0.3 ke 0.5 untuk mengurangi beban
    end
end)
